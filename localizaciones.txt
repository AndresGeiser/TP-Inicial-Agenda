CREATE TABLE IF NOT EXISTS countries (id int AUTO_INCREMENT NOT NULL,name varchar(100) NOT NULL,CONSTRAINT pk_countries PRIMARY KEY (id));
CREATE TABLE IF NOT EXISTS states (id int AUTO_INCREMENT NOT NULL,id_country int NOT NULL,name varchar(100) NOT NULL,CONSTRAINT pk_states PRIMARY KEY (id),CONSTRAINT fk_states_countries FOREIGN KEY (id_country) REFERENCES countries(id));
CREATE TABLE IF NOT EXISTS cities (id int AUTO_INCREMENT NOT NULL,id_state int NOT NULL,name varchar(100) NOT NULL,CONSTRAINT pk_cities PRIMARY KEY (id),CONSTRAINT fk_cities_states FOREIGN KEY (id_state) REFERENCES states(id));
